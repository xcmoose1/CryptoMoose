Smart Contract Example

// SPDX-License-Identifier: MIT pragma solidity ^0.8.0; import "@openzeppelin/contracts/token/ERC721/ERC721.sol"; import "@openzeppelin/contracts/access/Ownable.sol"; contract MyNFT is ERC721, Ownable { uint256 private _tokenIds; constructor() ERC721("MyNFT", "MNFT") {} function mint(address recipient, string memory tokenURI) public onlyOwner returns (uint256) { _tokenIds++; uint256 newItemId = _tokenIds; _mint(recipient, newItemId); _setTokenURI(newItemId, tokenURI); return newItemId; } }

💡 Pro Tips
